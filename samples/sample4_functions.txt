// Sample 4: Function Declarations
Matter add(Matter a, Matter b) {
    Resolve a + b;
}

Energy calculateArea(Energy radius) {
    Energy pi = 3.14159;
    Resolve pi * radius * radius;
}

Event main() {
    Matter result = add(5, 10);
    Energy area = calculateArea(2.5);
    Broadcast("Calculations complete");
    Resolve 0;
}
