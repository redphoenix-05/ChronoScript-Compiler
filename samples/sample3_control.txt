// Sample 3: Control Flow Structures
Event main() {
    Matter count = 5;
    
    Era (count > 0) {
        Broadcast("Count is positive");
    } Alternate {
        Broadcast("Count is not positive");
    }
    
    Loop (Matter i = 0; i < count; i = i + 1) {
        Broadcast("Iteration");
    }
    
    Resolve 0;
}
